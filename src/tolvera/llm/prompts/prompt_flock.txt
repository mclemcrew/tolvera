--- NEGATIVE EXAMPLE F1 (INCORRECT FORMAT - DO NOT DO THIS) ---
Request: "Flock sketch 'blue_vs_red', 500 particles, 2 species. Species 0 blue, species 1 red."
INCORRECT JSON (DO NOT OUTPUT THIS FORMAT):
{
  "name": "blue_vs_red",                  // ❌ WRONG: should be "sketch_name"
  "sketch_type": "flock",
  "particles": 500,                       // ❌ WRONG: should be "particle_count"
  "species_count": 2,
  "species": [                            // ❌ WRONG: should be "species_configs"
    { "id": 0, "color": "#0000FF" },      // ❌ WRONG: should be "species_index" and color as name string
    { "id": 1, "color": "#FF0000" }       // ❌ WRONG: should be "species_index" and color as name string
  ],
  "particle_shape": "circle",             // ❌ WRONG: should be inside "render_config"
  "render_config": {
    "background_behavior": "diffuse",
    "diffuse_rate": 0.95
  }
}

--- EXAMPLE F2 (Rules) ---
Request: "Flocking, 2 species, 200 particles, name 'rule_test'. Species 0 avoid species 1 with separation 0.8 radius 0.1."
INCORRECT JSON (DO NOT OUTPUT THIS FORMAT):
{
  "name": "rule_test",                  // ❌ WRONG: should be "sketch_name"
  "sketch_type": "flock",
  "particles": 200,                     // ❌ WRONG: should be "particle_count"
  "flock_config": {
    "module_name": "flock",
    "rules": [
      { "species_from": 0, "species_to": 1, "separate": 0.8, "radius": 0.1 }
    ]
  },
  "species": [                         // ❌ WRONG: should be "species_configs"
     { "id": 0, "count": 100, "color": { "r": 0.5, "g": 0.5, "b": 0.5 } },  // ❌ WRONG: should use string color names
     { "id": 1, "count": 100, "color": "#FFFFFF" }  // ❌ WRONG: should use string color names
  ]
}
